    ```tsx
    // components/ui/SocialIcon.tsx
    // Server Component — no "use client" directive
    // Per-platform inline SVG anchor. Renders null when url is empty/undefined (D-12).
    // 20×20 icon inside a 44×44 touch target via p-3 padding (D-13, WCAG 2.5.5 AAA).
    // Simple Icons CC0-1.0 license; viewBox="0 0 24 24" (D-10).
    // Source: RESEARCH Pitfall #7 (lines 625-681); CONTEXT D-10/D-11/D-12/D-13.

    export type Platform =
      | 'instagram'
      | 'soundcloud'
      | 'beatport'
      | 'youtube'
      | 'tiktok'
      | 'facebook';

    // Simple Icons CC0 SVG path d-attribute per platform.
    // Source: https://github.com/simple-icons/simple-icons/tree/develop/icons
    // Each string is the full contents of the <path d="..."> from the source SVG.
    const ICON_PATHS: Record<Platform, string> = {
      instagram: '{{ instagram_path }}',
      soundcloud: '{{ soundcloud_path }}',
      beatport: '{{ beatport_path }}',
      youtube: '{{ youtube_path }}',
      tiktok: '{{ tiktok_path }}',
      facebook: '{{ facebook_path }}',
    };

    const LABELS: Record<Platform, string> = {
      instagram: 'Instagram',
      soundcloud: 'SoundCloud',
      beatport: 'Beatport',
      youtube: 'YouTube',
      tiktok: 'TikTok',
      facebook: 'Facebook',
    };

    export default function SocialIcon({
      platform,
      url,
    }: {
      platform: Platform;
      url: string | null | undefined;
    }) {
      // D-12: render null when URL is empty/undefined — no broken link, no empty icon
      if (!url) return null;

      return (
        <a
          href={url}
          target="_blank"
          rel="noopener noreferrer"
          aria-label={`${LABELS[platform]} — Marginalia`}
          className="inline-flex items-center justify-center p-3 text-[--color-text-secondary] hover:text-[--color-accent-lime] transition-colors duration-150"
        >
          <svg
            width="20"
            height="20"
            viewBox="0 0 24 24"
            fill="currentColor"
            aria-hidden="true"
            xmlns="http://www.w3.org/2000/svg"
          >
            <path d={ICON_PATHS[platform]} />
          </svg>
        </a>
      );
    }
    ```

    Concrete values (do NOT substitute):
    - Icon size: `width="20" height="20"` — D-13 locks 20px

    - Default color: `text-[--color-text-secondary]` (#D2D2DB per UI-SPEC line 214)
    - Hover color: `hover:text-[--color-accent-lime]` (#9EFF0A per UI-SPEC line 215)

    - a11y: `aria-label="${LABELS[platform]} — Marginalia"` on anchor, `aria-hidden="true"` on decorative SVG
    - D-12: `if (!url) return null;` MUST be the first statement of the function body after the props destructure

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Server render → Browser | Inline SVG markup (Logo, SocialIcon) shipped in server HTML. Path data is hardcoded from Simple Icons (CC0) and the client's logo file — no user-derived input. |
| SocialIcon anchor → External origins (Instagram, etc.) | Outbound link; user's browser is the client, external origin is outside our trust boundary. |
| Logo component → public/logo.svg | Developer at execution time reads the SVG file and inlines its contents. Not a runtime boundary. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Tampering | SocialIcon external link | mitigate | Every anchor uses `target="_blank" rel="noopener noreferrer"` — prevents reverse-tabnabbing (ASVS V14.5). Enforced by acceptance criteria grep. |
| T-02-02-02 | Tampering | Inline SVG path data | accept | Simple Icons path strings are hardcoded CC0 assets, not user-derived. No `dangerouslySetInnerHTML` is used anywhere in this plan — SVG path values flow through React's `d={...}` prop which is treated as a string attribute, not HTML. |
| T-02-02-03 | Information Disclosure | Logo SVG file contents | accept | `public/logo.svg` is a public brand asset by design; it will be served at `/logo.svg` or inlined into every page. No secrets. |
| T-02-02-04 | Spoofing | SocialIcon href values | accept | Href values come from Keystatic siteConfig (edited by trusted label operators in v1 local mode). No public submission path. |
| T-02-02-05 | Elevation of Privilege | Components accidentally shipped as Client Components | mitigate | Plan enforces `grep -q "use client"` exits non-zero for all three files. Prevents accidental import of reader/server-only modules into client bundle. |
</threat_model>
