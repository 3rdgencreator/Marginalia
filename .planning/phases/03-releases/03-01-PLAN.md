    import 'server-only';
    import { reader } from './keystatic';

    /**
     * Build the SoundCloud iframe embed URL from a canonical premiere URL.
     *
     * Rationale (CONTEXT D-12, UI-SPEC §SoundCloud Embed Contract):
     *   - color=#9EFF0A → URL-encoded to %239EFF0A by URLSearchParams
     *   - visual=true → gives artwork-forward player that reads on dark bg
     *   - auto_play=false → never auto-play; user consent
     *   - hide_related=true, show_comments=false, show_reposts=false,
     *     show_teaser=false → quiet, focused player
     *   - show_user=true → credit the artist account
     *
     * Pitfall #8: if premiereUrl itself contains query params (e.g. ?in=label/...)
     * the `?` must be encoded as part of the value. URLSearchParams handles this.
     * Do NOT string-concatenate; do NOT manually encodeURIComponent.
     */
    export function buildSoundCloudEmbedUrl(premiereUrl: string): string {
      const params = new URLSearchParams({
        url: premiereUrl,
        color: '#9EFF0A',
        auto_play: 'false',
        hide_related: 'true',
        show_comments: 'false',
        show_user: 'true',
        show_reposts: 'false',
        show_teaser: 'false',
        visual: 'true',
      });
      return `https://w.soundcloud.com/player/?${params.toString()}`;
    }

    /**
     * Flatten a Keystatic document (array of block nodes with nested children
     * and text leaves) to a single plain-text string. Used for:
     *   - SEO meta descriptions (truncate to 160)
     *   - JSON-LD description field (truncate to 500)
     *
     * Returns a string with normalized whitespace, trimmed. No ellipsis is
     * appended — callers may append their own if desired.
     *
     * If document is not an array (e.g. description field is empty/unset on
     * a release entry), returns ''.
     */
    export function plainTextFromDocument(
      document: unknown,
      maxChars: number
    ): string {
      if (!Array.isArray(document)) return '';
      const walk = (node: unknown): string => {
        if (node && typeof node === 'object') {
          const n = node as { text?: unknown; children?: unknown };
          if (typeof n.text === 'string') return n.text;
          if (Array.isArray(n.children)) return n.children.map(walk).join('');
        }
        return '';
      };
      const flat = document.map(walk).join(' ').replace(/\s+/g, ' ').trim();
      return flat.length > maxChars ? flat.slice(0, maxChars) : flat;
    }

    /**
     * Translate a genre slug (kebab-case value stored in Keystatic) to the
     * human-readable label. Used for:
     *   - Genre chips on detail page (UI-SPEC genre chip mapping table)
     *   - JSON-LD `genre` array (Pattern 7)
     *
     * Mapping covers the 5 schema values locked in keystatic.config.ts. Any
     * future genre added to the schema must be added here too.
     */
    const GENRE_LABELS: Record<string, string> = {
      'melodic-house': 'Melodic House',
      techno: 'Techno',
      'indie-dance': 'Indie Dance',
      'organic-house': 'Organic House',
      'afro-house': 'Afro House',
    };

    export function genreLabelFor(slug: string): string {
      return GENRE_LABELS[slug] ?? slug;
    }

    /**
     * Resolve artist slugs from a release entry to display names by reading
     * the artists collection.
     *
     * Pitfall #6: artist entries are authored in Phase 4. Until then,
     * reader.collections.artists.read(slug) returns null for most slugs.
     * Fall back to the slug itself so the release detail page still renders
     * (degraded display — "elif" instead of "ELIF") rather than 500ing.
     *
     * Returns names in the same order as the input slugs, so `.join(' × ')`
     * respects authoring order (which is significant for credit sequencing).
     *
     * NOTE: artists.read returns an entry whose `name` field is the slug
     * (because `name` is the slugField in keystatic.config.ts). The
     * human-display name IS the slug text — there is no separate display-name
     * field on the artists schema. This helper therefore mostly returns the
     * slug-cased value either way; it exists so Phase 4 can later add a
     * `displayName` text field without refactoring every release renderer.
     */
    export async function resolveArtistNames(
      artistSlugs: readonly string[]
    ): Promise<string[]> {
      if (!artistSlugs || artistSlugs.length === 0) return [];
      const resolved = await Promise.all(
        artistSlugs.map(async (slug) => {
          try {
            const artist = await reader.collections.artists.read(slug);
            // Artists collection's slug field is `name`. If a future Phase 4
            // change adds a displayName text field, prefer it; today, fall
            // back to the slug.
            if (artist) {
              const displayName = (artist as unknown as { displayName?: string })
                .displayName;
              return displayName ?? slug;
            }
            return slug;
          } catch {
            return slug;
          }
        })
      );
      return resolved;
    }

| `next.config.ts` config → browser | Static config affecting `<img>` src formation. No runtime user input. |

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01-01 | Tampering | `buildSoundCloudEmbedUrl` URL param concatenation | mitigate | Use `URLSearchParams` (not manual encodeURIComponent + string concat) to correctly encode premiere URLs that already contain query strings.
