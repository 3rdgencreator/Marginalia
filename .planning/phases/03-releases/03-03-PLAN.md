    ```tsx
    import ArtistLink from './ArtistLink';

    type MetaHeaderProps = {
      title: string;
      artistSlugs: readonly string[];
      artistNames: readonly string[];
      releaseDate: string | null;
    };

    /**
     * Metadata header block at the top of the release detail page.
     *
     * Per D-09: "minimal — title, artist name(s), release date."
     * Catalog number, genres, release type are RENDERED ELSEWHERE (Plan 04
     * puts them in a secondary section below the description).
     *
     * Structure (UI-SPEC §Release Detail Page / D-09 / lines 222-224):
     *   H1 (title, --text-heading on mobile / 2rem on desktop, tracking-tight, bold)
     *   Artist links joined by " × " (16px body, weight 400)
     *   Release date formatted as human-readable (Label size, muted)
     *
     * 16px vertical rhythm between lines (UI-SPEC line 224).
     */
    export default function ReleaseMetaHeader({
      title,
      artistSlugs,
      artistNames,
      releaseDate,
    }: MetaHeaderProps) {
      // Format YYYY-MM-DD as "D Month YYYY" for display (e.g. "15 March 2025").
      // Use en-GB locale to match UK-English "15 March 2025" order; the label
      // is Barcelona-based but the site ships in English and this order is
      // unambiguous vs. MM/DD/YYYY or DD/MM/YYYY numeric formats.
      const formattedDate = releaseDate
        ? new Date(releaseDate + 'T00:00:00Z').toLocaleDateString('en-GB', {
            day: 'numeric',
            month: 'long',
            year: 'numeric',
            timeZone: 'UTC',
          })
        : null;

      return (
        <header className="flex flex-col gap-4">
          <h1 className="text-[--text-heading] md:text-[2rem] font-bold tracking-[-0.02em] leading-tight text-[--color-text-primary]">
            {title}
          </h1>
          {artistSlugs.length > 0 && (
            <p className="text-[--text-body] text-[--color-text-primary]">
              {artistSlugs.map((slug, index) => {
                const name = artistNames[index] ?? slug;
                return (
                  <span key={slug}>
                    <ArtistLink slug={slug} name={name} />
                    {index < artistSlugs.length - 1 && (
                      <span className="text-[--color-text-secondary]"> × </span>
                    )}
                  </span>
                );
              })}
            </p>
          )}
          {formattedDate && (
            <time
              dateTime={releaseDate ?? undefined}
              className="text-[--text-label] text-[--color-text-secondary]"
            >
              Out {formattedDate}
            </time>
          )}
        </header>
      );
    }
    ```

    ```tsx
     * Laylo presave/save CTA (D-16).
     *

    export default function LayloButton({

      releaseTitle,
      releaseDate,
    }: LayloButtonProps) {
      if (!url) return null;

      // "Save" if the release is already out (releaseDate <= today).
      // Compare YYYY-MM-DD strings lexicographically, null = "Pre-save".
      const today = new Date().toISOString().slice(0, 10);

      const verb = isOut ? 'Save' : 'Pre-save';

      return (
        <a
          href={url}
          target="_blank"
          rel="noopener noreferrer"
          aria-label={`${verb} "${releaseTitle}" on Laylo`}
          className="
            inline-flex items-center justify-center
            rounded-md px-6 py-3
            bg-[--color-accent-lime] text-[--color-bg] font-bold
            hover:bg-[--color-accent-violet] hover:text-[--color-text-primary]
            transition-colors duration-150
            text-[--text-body]
          "
        >
          {label}
        </a>
      );
    }
    ```

    ```tsx
     * Summary label: "Also available on" (UI-SPEC line 402).
     *
     * NOTE: the native <details> disclosure triangle (::-webkit-details-marker

     * default marker and renders an explicit CSS caret via ::before on the
     * summary — guarantees cross-browser consistency (RESEARCH Open Question
    ```

    **Notes on the <details> marker suppression:**
    - `list-style: none` on `<summary>` hides Firefox's default disc marker.
    - `::-webkit-details-marker { display: none }` is required for Safari. Tailwind v4 supports the `[&_summary::-webkit-details-marker]:hidden` arbitrary-variant syntax.
    - Chrome respects `list-style: none` on `<summary>` as well.
    - The custom SVG caret rotates 90° on open via `group-open:rotate-90` (Tailwind `group` + `open` attribute on `<details>`).

    Do NOT include `open` attribute on `<details>` — UI-SPEC requires default-collapsed state (Pitfall for disclosure authoring).
